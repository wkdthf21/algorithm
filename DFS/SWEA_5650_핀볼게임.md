```c++
#include<iostream>
#include<vector>
#include<memory.h>
using namespace std;

int N, maxScore, startX, startY;
int arr[101][101];
pair<int, int> direction[] = { {-1, 0}, {1, 0}, {0, -1}, {0, 1} };
vector<pair<int, int>> holeVec[5];
vector<pair<int, int>> startVec;

int search(int y, int x, int directionNum){
    
    int score = 0;

    while(1){

		int ny = y;
    	int nx = x;
		int nd = directionNum;

        // 도착여부 검사
        if(arr[ny][nx] == -1 || (ny == startY && nx == startX) ){
            if(score > maxScore) maxScore = score;
            break;
        }

        // 블록 검사
        int value = arr[ny][nx];
        if(value == 1){
            if(nd == 0) nd = 1;
            else if(nd == 1) nd = 3;
            else if(nd == 2) nd = 0;
            else if(nd == 3) nd = 2;
            ++score;
        }
        else if(value == 2){
            if(nd == 0) nd = 3;
            else if(nd == 1) nd = 0;
            else if(nd == 2) nd = 1;
            else if(nd == 3) nd = 2;
            ++score;
        }
        else if(value == 3){
            if(nd == 0) nd = 2;
            else if(nd == 1) nd = 0;
            else if(nd == 2) nd = 3;
            else if(nd == 3) nd = 1;
            ++score;
        }
        else if(value == 4){
            if(nd == 0) nd = 1;
            else if(nd== 1) nd = 2;
            else if(nd == 2) nd = 3;
            else if(nd == 3) nd = 0;
            ++score;
        }
        else if(value == 5){
            if(nd == 0) nd = 1;
            else if(nd == 1) nd = 0;
            else if(nd == 2) nd = 3;
            else if(nd == 3) nd = 2;
            ++score;
        }
        // 웜홀 검사    
        else if(value >=  6 && value <= 10){
            int holeY, holeX;
            for(int i = 0; i < 2; i++){
                pair<int, int> hole = holeVec[value - 6][i];
                holeY = hole.first;
                holeX = hole.second;
                if( holeY == ny && holeX == nx ) continue;
                // 짝지어진 웜홀로 이동
                else break;
            }
            ny = holeY; nx = holeX;
        }

        y = ny + direction[nd].first;
        x = nx + direction[nd].second;
		directionNum = nd;

        if(y < 0 || y > N || x < 0 || x > N) break;

    }

    return score;
}

int main(int argc, char** argv)
{
	int test_case;
	int T;
	scanf("%d", &T);
	for(test_case = 1; test_case <= T; ++test_case)
	{
        // initialize
        memset(arr, 0, sizeof(arr));
        maxScore = 0, startX = 0, startY = 0;
        for(int i = 0; i < 5; i++)
            holeVec[i].clear();
        startVec.clear();

        // 벽 처리
        // map바깥은 블럭 5로 둘러싸준다.
       for (int i = 0; i <= N+1; i++) {
			arr[0][i] = 5;
			arr[i][0] = 5;
			arr[N + 1][i] = 5;
			arr[i][N + 1] = 5;
		}

        // input
		scanf("%d", &N);
        for(int i = 1; i <= N; i++){
            for(int j = 1; j <= N; j++){
                scanf("%d", &arr[i][j]);
                if(arr[i][j] >= 6 && arr[i][j] <= 10) holeVec[arr[i][j] - 6].push_back(make_pair(i, j)); // 웜홀
                else if(arr[i][j] == 0) startVec.push_back({i, j});
            }
        }

        // start
        for(auto iter = startVec.begin(); iter != startVec.end(); iter++){
            int y = iter->first;
            int x = iter->second;
            for(int k = 0; k < 4; k++){       
                int newY = y + direction[k].first;        
                int newX = x + direction[k].second;
                if(newY < 0 || newY > N || newX < 0 || newX > N) continue;
                startY = y; startX = x;
                int score = search(newY, newX, k);
                if(maxScore < score) maxScore = score;
            }
       	}  

        printf("#%d %d \n", test_case, maxScore);
    }
	return 0;//정상종료시 반드시 0을 리턴해야합니다.
}
```
