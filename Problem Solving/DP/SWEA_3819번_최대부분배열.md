```c++
#include <iostream>
#include <algorithm>

using namespace std;

int arr[200000];

int main(int argc, char** argv)
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	int test_case;
	int T;

	cin >> T;

	int N, ans;
	int temp;
	for (test_case = 1; test_case <= T; ++test_case)
	{
		cin >> N;
		for (int i = 0; i < N; i++)
			cin >> arr[i];

		temp = arr[0];
		ans = arr[0];
		for (int i = 1; i < N; i++) {
			temp = max(0, temp) + arr[i];
			ans = max(ans, temp);
		}

		cout << "#" << test_case << " " << ans << "\n";

	}
	return 0;
}
```
